232  //
233  // two staged computation task
234  // this is used when a task handle is returned when a call is made,
235  // while the task, is however, enqueued later after other operations when
236  // certain parameters to the task is known (e.g., error code after logging)
237  // in thise case, we can use two staged computation task as this is.
238  //
239  // task_ptr task = tasking::create_late_task(...);
240  // ...
241  // return task;
242  //
243  // ... after logging ....
244  // task->bind_and_enqueue([&](cb)=>{std::bind(cb, error)}, delay);
245  //
246  template<typename THandler>
247  class safe_late_task : public safe_task_handle
248  {
249  public:
250  safe_late_task(THandler& h)
251  : _handler(h), _bound_handler(nullptr)
252  {
253  }
254 
255  operator task_ptr() const
256  {
257  return task_ptr(this);
258  }
259 
260  virtual bool cancel(bool wait_until_finished, bool* finished = nullptr) override
261  {
262  bool r = safe_task_handle::cancel(wait_until_finished, finished);
263  if (r)
264  {
265  _bound_handler = nullptr;
266  }
267  return r;
268  }
269 
270  void bind_and_enqueue(
271  std::function<std::function<void()> (THandler&)> binder,
272  int delay_milliseconds = 0
273  )
274  {
275  _bound_handler = binder(_handler);
276  _handler = nullptr;
277  dsn_task_call(native_handle(), delay_milliseconds);
278  }
279 
280  static void on_cancel(void* task)
281  {
282  auto t = (safe_late_task<THandler>*)task;
283  t->release_ref(); // added upon callback exec registration
284  }
285 
286  static void exec(void* task)
287  {
288  auto t = (safe_late_task<THandler>*)task;
289  t->_bound_handler();
290  t->_bound_handler = nullptr;
291  t->release_ref(); // added upon callback exec registration
292  }
293 
294  private:
295  std::function<void()> _bound_handler;
296  THandler _handler;
297  };
